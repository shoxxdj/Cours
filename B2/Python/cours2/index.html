class Human:
	def doStuff(self):
		print "I do Stuff"

	def __init__(self,prenom,nom):
		self.nom = nom
		self.prenom = prenom

class Warrior(Human):
	def kamehameha(self):
		print "I can shot you over %s"%(self.force)

	def __init__(self,prenom,nom,force):
		Human.__init__(self,prenom,nom) #Rappel du constructeur
		self.force = force


michelsan = Human("michel","san")
michelsan.doStuff()

songo = Warrior("son","goku",9000)
songo.kamehameha()
